    __init__(self, db_path: str = "knowledge/legion.db"):

    _init_db():
      - Execute PRAGMA journal_mode=WAL
      - Execute PRAGMA busy_timeout=5000
      - Execute PRAGMA foreign_keys=ON
      - CREATE TABLE IF NOT EXISTS content (
          id INTEGER PRIMARY KEY AUTOINCREMENT,
          source_type TEXT NOT NULL,
          source_url TEXT,
          title TEXT NOT NULL,
          raw_content TEXT NOT NULL,
          processed_date TEXT NOT NULL,
          content_hash TEXT UNIQUE NOT NULL,
          reference_count INTEGER DEFAULT 1
        )
      - CREATE TABLE IF NOT EXISTS topics (
          id INTEGER PRIMARY KEY AUTOINCREMENT,
          name TEXT NOT NULL UNIQUE,
          slug TEXT NOT NULL UNIQUE,
          created_at TEXT NOT NULL
        )
      - CREATE TABLE IF NOT EXISTS content_topics (
          content_id INTEGER REFERENCES content(id) ON DELETE CASCADE,
          topic_id INTEGER REFERENCES topics(id) ON DELETE CASCADE,
          PRIMARY KEY (content_id, topic_id)
        )
      - CREATE INDEX IF NOT EXISTS idx_content_hash ON content(content_hash)

    async def insert_content(item: ContentItem) -> int:
      - INSERT OR IGNORE INTO content (source_type, source_url, title, raw_content, processed_date, content_hash, reference_count)
        VALUES (item.source_type, item.source_url, item.title, item.raw_content, item.processed_date, item.content_hash, 1)
      - Check cursor.rowcount: if 0, content_hash already exists
        - UPDATE content SET reference_count = reference_count + 1 WHERE content_hash = item.content_hash
        - SELECT id FROM content WHERE content_hash = item.content_hash
      - Return the content id

    async def get_content(content_id: int) -> ContentItem | None

    async def get_all_content() -> list[ContentItem]

    async def content_exists_by_hash(content_hash: str) -> bool:
      - Returns True if a content row with this hash exists

    async def get_or_create_topic(name: str) -> tuple[int, str]:
      - Slugify name: slug = re.sub(r'[^a-z0-9]+', '-', name.lower()).strip('-')
      - INSERT OR IGNORE into topics
      - Return (topic_id, slug)

    async def link_content_to_topic(content_id: int, topic_id: int) -> None

    async def get_content_topics(content_id: int) -> list[str]:
      - Returns list of topic names for a content item

    async def get_topic_content(topic_slug: str) -> list[ContentItem]

    async def close()
