    async def show_summary_menu(content_id: int, title: str) -> str:
      - SUMMARY-01: 4-option quick-select menu
      - Use questionary.select() with .ask_async()
      - Display: f"Content processed: **{title}**\n\nWhat would you like to do?"
      - choices = [
          "Quick Summary",      # SUMMARY-02: 3-5 bullet points
          "Full Breakdown",     # SUMMARY-03: timestamps, key quotes, follow-up questions
          "Ask a Question",    # SUMMARY-04: user types question
          "Save for Later",     # SUMMARY-05: defer processing
        ]
      - Return the selected option string

    async def quick_summary(content_id: int, db: Database) -> str:
      - SUMMARY-02: Returns 3-5 key points in bullet format
      - content = await db.get_content(content_id)
      - if not content: raise ValueError(f"Content {content_id} not found")
      - Split raw_content into sentences (split on '. ' or '\n')
      - Filter to sentences with 10+ chars, remove boilerplate
      - Return first 3-5 as bullets: "- " + sentence.strip()
      - Format as markdown bullet list
      - If content is very short (< 50 chars), return content.raw_content itself

    async def full_breakdown(content_id: int, db: Database) -> str:
      - SUMMARY-03: Returns structured notes
      - content = await db.get_content(content_id)
      - if not content: raise ValueError(f"Content {content_id} not found")
      - Build markdown output:
        - ## Key Moments (for youtube/transcript content)
          - Try to parse transcript for timestamps if available
          - Format as "- [HH:MM:SS] Moment description"
          - If no timestamps, show first 3 significant sentences
        - ## Key Quotes
          - Extract 2-3 sentences with ! ? or quotes
          - Format as "> quote text"
        - ## Follow-up Questions
          - Generate 2-3 questions based on content keywords
          - Format as "- Question text"
      - Return formatted markdown string

    async def ask_question_mode(content_id: int, db: Database, question: str) -> str:
      - SUMMARY-04: Answer a user question about the content
      - content = await db.get_content(content_id)
      - if not content: raise ValueError(f"Content {content_id} not found")
      - Extract keywords from question (words with 4+ chars)
      - Search raw_content for sentences containing any keyword (case-insensitive)
      - Return top 3 matches as:
        "Based on the content:\n> [matching sentence]\n\n"
      - If no matches found, return: "I couldn't find anything about that in the content. Could you rephrase or ask about something more specific?"

    async def save_for_later(content_id: int, db: Database) -> str:
      - SUMMARY-05: Mark content as deferred
      - content = await db.get_content(content_id)
      - if not content: raise ValueError(f"Content {content_id} not found")
      - For Phase 1: just confirm with user
      - Return: f"**Saved for later:** {content.title}\n\nUse 'what do I have on X?' to retrieve this content anytime."
