    def detect_content_type(input_str: str) -> str:
      - Returns: 'youtube', 'article', 'pdf', or 'text'
      - Strip whitespace from input with input_str.strip()
      - If starts with 'http://' or 'https://':
        - Parse URL with urllib.parse.urlparse
        - Check netloc for 'youtube.com', 'youtu.be' -> return 'youtube'
        - Check path.lower().endswith('.pdf') -> return 'pdf'
        - Otherwise -> return 'article'
      - If len(input_str) > 1000 and '\x00' in input_str -> return 'pdf'
      - If '\n' in input_str and len(input_str) > 100 -> return 'text'
      - If len(input_str) < 500 and not input_str.startswith('http') -> return 'text'
      - Default -> return 'text'

    def extract_video_id(youtube_url: str) -> str:
      - Handle youtube.com/watch?v=ID format:
        - parsed = urlparse, parsed.query contains 'v=ID'
      - Handle youtu.be/ID format:
        - path.split('/')[-1]
      - Return the video ID string
      - Raise ValueError if neither format matches

    Create src/pipeline/extractors/youtube_extractor.py:
    - async def extract_youtube(url: str) -> tuple[str, str, str]:
      - Returns (title, raw_transcript, video_id)
      - video_id = extract_video_id(url)
      - transcript = youtube_transcript_api.YouTubeTranscriptApi.get_transcript(video_id)
      - raw_transcript = " ".join(entry["text"] for entry in transcript)
      - title from video_id (fetch via requests to oembed or use f"YouTube Video {video_id}")
      - Handle exceptions: if transcript unavailable, return (title, "[Transcript unavailable - video may not have captions]", video_id) with warning logged via logging.warning()
      - If video_id extraction fails, raise ValueError with message "Invalid YouTube URL"

    Create src/pipeline/extractors/article_scraper.py:
    - async def scrape_article(url: str) -> tuple[str, str]:
      - Returns (title, main_text)
      - response = requests.get(url, timeout=10, headers={"User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36"})
      - soup = BeautifulSoup(response.text, "html.parser")
      - title = (soup.find("meta", property="og:title") or soup.find("title")).get("content", None) or soup.find("title").text.strip() if soup.find("title") else "Untitled"
      - article = (soup.find("article") or soup.find("main") or soup.find(class_=re.compile(r"content|article|post", re.I)))
      - text = article.get_text(separator="\n", strip=True) if article else response.text[:500]
      - If len(text) < 200 or any(kw in text.lower() for kw in ["nav", "menu", "subscribe"]): log warning
      - Return (title, text)

    Create src/pipeline/extractors/pdf_extractor.py:
    - async def extract_pdf(source: str | bytes) -> tuple[str, str]:
      - If source is URL string starting with http, download with requests.get
      - If source is bytes, wrap in BytesIO
      - doc = fitz.open(stream=source, filetype="pdf") if isinstance(source, bytes) else fitz.open(source)
      - text_parts = [page.get_text() for page in doc]
      - doc.close()
      - text = "\n".join(text_parts)
      - title from doc.metadata.get("title") or "Untitled PDF"
      - If not text.strip(): return (title, "[No text extracted from PDF]")
      - Return (title, text)

    Create src/pipeline/extractors/text_classifier.py:
    - async def classify_and_store(text: str) -> tuple[str, str]:
      - Returns (title, processed_text)
      - Strip text with text.strip()
      - Collapse multiple blank lines: processed = re.sub(r'\n{3,}', '\n\n', text)
      - title = first line of text[:100] or "Untitled Text"
      - Return ('text', title, processed)

    Create src/pipeline/extractors/__init__.py exporting all extractors.

    - async def process_content(input_str: str, sync_manager: SyncManager, topic: str = "general") -> ContentItem:
      - content_type = detect_content_type(input_str)
      - Route:
        - 'youtube': title, raw_content, _ = await extract_youtube(input_str)
        - 'article': title, raw_content = await scrape_article(input_str)
        - 'pdf': title, raw_content = await extract_pdf(input_str)
        - 'text': _, title, raw_content = await classify_and_store(input_str)
      - Build ContentItem:
        - source_type = content_type
        - source_url = input_str if content_type in ('youtube', 'article', 'pdf') else None
        - title = title
        - raw_content = raw_content
        - processed_date = datetime.now().isoformat()
        - content_hash = compute_content_hash(raw_content)
        - reference_count = 1
      - content_id = await sync_manager.write_content(item, topic)
      - item.id = content_id
      - Return item
